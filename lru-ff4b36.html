LRU的思想：将最近一次使用的时间作为衡量某个资源（如内存）的价值，最近一次使用的时间越久，它的价值就越小。

即淘汰最近一次使用时间最长的数据。最久未使用的数据

手写双向链表➕HashMap

public LRUCache{
//双向链表NODE节点
	public Node{
				String key;
        int value;
        DLinkedNode pre;
        DLinkedNode post;
  }
//存储具体的数据
	private ConcurrentMap<String, DLinkedNode> cache = new ConcurrentHashMap<String, DLinkedNode>();
	private int count;
	private int capacity;
	//记录双向链表的头节点和尾节点
  private DLinkedNode head, tail;
//构造方法，初始化头尾节点
	public LRUCache(int capacity) {
        this.count = 0;
        this.capacity = capacity;

        head = new DLinkedNode();
        head.pre = null;

        tail = new DLinkedNode();
        tail.post = null;

        head.post = tail;
        tail.pre = head;
    }
//工具方法，将某个节点添加到头部
	private void addNode(DLinkedNode node){
        node.pre = head;
        node.post = head.post;

        head.post.pre = node;
        head.post = node;
    }
//工具方法 删除某个节点
    private void removeNode(DLinkedNode node){
        DLinkedNode pre = node.pre;
        DLinkedNode post = node.post;

        pre.post = post;
        post.pre = pre;
    }
//工具方法 移动某个节点到头部
    private void moveToHead(DLinkedNode node){
        removeNode(node);
        addNode(node);
    }
//工具方法 删除尾部节点
    private DLinkedNode popTail(){
        DLinkedNode res = tail.pre;
        removeNode(res);
        return res;
    }
//get方法，会更新双向链表中的数据
	public int get(String key) {
        DLinkedNode node = cache.get(key);
        if(node == null){
            return -1; // should raise exception here.
        }
        moveToHead(node);
        return node.value;
    }

//put方法，会检查容量值
    public void put(String key, int value) {
        DLinkedNode node = cache.get(key);
        if (node != null) {
            node.value = value;
            moveToHead(node);
            return;
        }
        DLinkedNode newNode = new DLinkedNode();
        newNode.key = key;
        newNode.value = value;

        cache.put(key, newNode);
        addNode(newNode);

        ++count;

        if(count > capacity){
            // pop the tail
            DLinkedNode tail = popTail();
            cache.remove(tail.key);
            --count;
        }
    }

}

继承LInkedHashMap的方法

public class LinkedListLRU extends LinkedHashMap {
    private int capacity;
		/**
     * 当LinkedHashMap的accessOrder参数为true时，即会按照访问顺序排序，最近访问的放在最前，最早访问的放在后面
     */
    public LinkedListLRU(int capacity){
        super(capacity, 0.75f, true);
        this.capacity = capacity;
    }
		/**
     * LinkedHashMap自带的判断是否删除最老的元素方法，默认返回false，即不删除老数据
     */
    @Override
    protected boolean removeEldestEntry(Map.Entry eldest) {
        return size()>capacity;
    }
}
